40/// `GET /api/onboard/catalog` — list every model provider the CLI wizard knows
41/// about. The dashboard shows these in the "+ Add model provider" picker so
42/// CLI / web stay in sync.
43pub async fn handle_catalog(State(state): State<AppState>, headers: HeaderMap) -> Response {
44    if let Err(e) = require_auth(&state, &headers) {
45        return e.into_response();
46    }
47    let _ = state;
48
49    let model_providers: Vec<CatalogModelProvider> = zeroclaw_providers::list_model_providers()
50        .into_iter()
51        .map(|p| CatalogModelProvider {
52            name: p.name.to_string(),
53            display_name: p.display_name.to_string(),
54            local: p.local,
55        })
56        .collect();
57
58    axum::Json(CatalogResponse { model_providers }).into_response()
59}

84/// `GET /api/onboard/catalog/models?model_provider=<name>` — fetch the model list
85/// for one model_provider. Same code path the CLI wizard uses
86/// (`zeroclaw_providers::create_model_provider(...).list_models()`), which goes
87/// through the models.dev cached catalog for OpenAI / Anthropic / Gemini,
88/// the live `/v1/models` endpoint for OpenRouter, etc.
89///
90/// Lazy: the dashboard hits this only when the user picks a model_provider, so
91/// initial catalog load stays fast. Fetch failures return an empty list
92/// with `live: false` so the form falls back to a free-text input.
93pub async fn handle_catalog_models(
94    State(state): State<AppState>,
95    headers: HeaderMap,
96    Query(q): Query<ModelsQuery>,
97) -> Response {
98    if let Err(e) = require_auth(&state, &headers) {
99        return e.into_response();
100    }
101    let _ = state;
102    let local = model_provider_family_is_local(&q.model_provider);
103
104    // Try provider construction + list_models first (covers credentialed
105    // /models endpoints). Fall back to the family catalog table when
106    // construction or list_models fails — this is the path that covers
107    // onboard mode where the operator hasn't supplied a credential
108    // and the provider has typed required fields (Azure resource,
109    // Bedrock region, …) that haven't been populated yet.
110    let provider_path = match zeroclaw_providers::create_model_provider(&q.model_provider, None) {
111        Ok(h) => match h.list_models().await {
112            Ok(ms) if !ms.is_empty() => Some(ms),
113            _ => None,
114        },
115        Err(_) => None,
116    };
117
118    let (models, live) = match provider_path {
119        Some(ms) => (ms, true),
120        None => {
121            match zeroclaw_providers::catalog::list_models_for_family(&q.model_provider).await {
122                Ok(ms) => (ms, true),
123                Err(e) => {
124                    ::zeroclaw_log::record!(DEBUG, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_attrs(::serde_json::json!({"model_provider": q.model_provider, "error": format!("{}", e)})), "model catalog fetch failed");
125                    (Vec::new(), false)
126                }
127            }
128        }
129    };
130
131    axum::Json(ModelsResponse {
132        model_provider: q.model_provider,
133        models,
134        local,
135        live,
136    })
137    .into_response()
138}

200pub struct OnboardStatusResponse {
201    /// `true` when no agent is dispatchable yet. The dashboard uses this
202    /// signal to redirect first-load visits from `/` to `/onboard`.
203    pub needs_onboarding: bool,
204    /// Short machine-readable reason for the value of `needs_onboarding`,
205    /// for logs / debugging. Stable: `fresh_install` / `incomplete_agent`
206    /// / `has_dispatchable_agent`.
207    pub reason: &'static str,
208    /// `true` when the operator has started entering setup state even if no
209    /// agent can reply yet. The dashboard uses this to say "Continue
210    /// onboarding" instead of pretending the flow is fresh.
211    pub has_partial_state: bool,
212    /// Human-readable readiness failures. When onboarding cannot finish, the
213    /// UI shows these directly so the operator knows exactly what is missing.
214    pub missing: Vec<String>,
215}

217/// Pure derivation of the onboard-status response from a config snapshot.
218/// `needs_onboarding` is `false` iff at least one enabled `[agents.<alias>]`
219/// block has a resolved model provider with a selected model plus resolved
220/// risk/runtime profile refs. A provider without a bound, runnable agent is
221/// not a completion signal: chat dispatch still bounces with a setup error in
222/// that state.
223#[must_use]
224pub fn derive_onboard_status(cfg: &zeroclaw_config::schema::Config) -> OnboardStatusResponse {
225    let missing = onboard_missing_requirements(cfg);
226    let ready = missing.is_empty();
227    let has_partial_state = !cfg.onboard_state.completed_sections.is_empty()
228        || cfg.providers.models.iter_entries().next().is_some()
229        || !cfg.risk_profiles.is_empty()
230        || !cfg.runtime_profiles.is_empty()
231        || !cfg.agents.is_empty();
232    let reason = if ready {
233        "has_dispatchable_agent"
234    } else if has_partial_state {
235        "incomplete_agent"
236    } else {
237        "fresh_install"
238    };
239    OnboardStatusResponse {
240        needs_onboarding: !ready,
241        reason,
242        has_partial_state,
243        missing,
244    }
245}

247fn onboard_missing_requirements(cfg: &zeroclaw_config::schema::Config) -> Vec<String> {
248    let mut missing = Vec::new();
249    if cfg.providers.models.iter_entries().next().is_none() {
250        missing.push("Add a model provider.".to_string());
251    }
252    if cfg.agents.is_empty() {
253        missing.push("Create an agent.".to_string());
254        return missing;
255    }
256
257    let mut agent_aliases: Vec<&String> = cfg.agents.keys().collect();
258    agent_aliases.sort();
259    let mut has_dispatchable_agent = false;
260    for alias in agent_aliases {
261        let agent_missing = onboard_agent_missing_requirements(cfg, alias, &cfg.agents[alias]);
262        if agent_missing.is_empty() {
263            has_dispatchable_agent = true;
264            break;
265        }
266        missing.extend(agent_missing);
267    }
268    if has_dispatchable_agent {
269        missing.clear();
270    }
271    missing
272}

274fn onboard_agent_missing_requirements(
275    cfg: &zeroclaw_config::schema::Config,
276    alias: &str,
277    agent: &zeroclaw_config::schema::AliasedAgentConfig,
278) -> Vec<String> {
279    let mut missing = Vec::new();
280    if !agent.enabled {
281        missing.push(format!("Enable agent `{alias}`."));
282    }
283
284    let model_ref = agent.model_provider.trim();
285    if model_ref.is_empty() {
286        missing.push(format!("Set a model provider for agent `{alias}`."));
287    } else if let Some((family, _, provider)) = cfg.resolved_model_provider_for_agent(alias) {
288        let has_model = provider
289            .model
290            .as_deref()
291            .map(str::trim)
292            .is_some_and(|m| !m.is_empty());
293        if !has_model {
294            missing.push(format!("Choose a model for model provider `{model_ref}`."));
295        } else if !model_provider_alias_usable(provider, model_provider_family_is_local(family)) {
296            missing.push(format!(
297                "Set credential/auth for model provider `{model_ref}`."
298            ));
299        }
300    } else {
301        missing.push(format!(
302            "Fix agent `{alias}` model provider `{model_ref}`; it does not resolve to a configured provider."
303        ));
304    }
305
306    let risk_ref = agent.risk_profile.trim();
307    if risk_ref.is_empty() {
308        missing.push(format!("Set a risk profile for agent `{alias}`."));
309    } else if !cfg.risk_profiles.contains_key(risk_ref) {
310        missing.push(format!(
311            "Fix agent `{alias}` risk profile `{risk_ref}`; it does not resolve to a configured profile."
312        ));
313    }
314
315    let runtime_ref = agent.runtime_profile.trim();
316    if runtime_ref.is_empty() {
317        missing.push(format!("Set a runtime profile for agent `{alias}`."));
318    } else if !cfg.runtime_profiles.contains_key(runtime_ref) {
319        missing.push(format!(
320            "Fix agent `{alias}` runtime profile `{runtime_ref}`; it does not resolve to a configured profile."
321        ));
322    }
323
324    missing
325}

360/// Build the `AgentOptionsResponse` from a config snapshot. Pure function
361/// so tests can drive the same code path the handler runs without spinning
362/// up an `AppState`.
363///
364/// `get_map_keys` expects **kebab-case** paths (the macro at
365/// `crates/zeroclaw-macros/src/lib.rs:366` builds lookup arms with
366/// `snake_to_kebab(field_name)`). Passing snake_case for any
367/// underscore-bearing field silently returns `None` → empty `Vec` →
368/// dashboard renders "No X configured yet" even though X is configured.
369pub fn build_agent_options(cfg: &zeroclaw_config::schema::Config) -> AgentOptionsResponse {
370    fn dotted_aliases(cfg: &zeroclaw_config::schema::Config, prefix: &str) -> Vec<String> {
371        let mut out: Vec<String> = Vec::new();
372        for f in cfg.prop_fields() {
373            if let Some(rest) = f.name.strip_prefix(&format!("{prefix}.")) {
374                let mut parts = rest.splitn(3, '.');
375                if let (Some(ty), Some(alias), Some(_)) = (parts.next(), parts.next(), parts.next())
376                {
377                    let dotted = format!("{ty}.{alias}");
378                    if !out.contains(&dotted) {
379                        out.push(dotted);
380                    }
381                }
382            }
383        }
384        out.sort();
385        out
386    }
387
388    let channels = dotted_aliases(cfg, "channels");
389    let mut channel_types: Vec<String> = channels
390        .iter()
391        .filter_map(|d| d.split_once('.').map(|(t, _)| t.to_string()))
392        .collect();
393    channel_types.sort();
394    channel_types.dedup();
395
396    AgentOptionsResponse {
397        channels,
398        channel_types,
399        model_providers: dotted_aliases(cfg, "providers.models"),
400        risk_profiles: cfg.get_map_keys("risk-profiles").unwrap_or_default(),
401        runtime_profiles: cfg.get_map_keys("runtime-profiles").unwrap_or_default(),
402        skill_bundles: cfg.get_map_keys("skill-bundles").unwrap_or_default(),
403        knowledge_bundles: cfg.get_map_keys("knowledge-bundles").unwrap_or_default(),
404        mcp_bundles: cfg.get_map_keys("mcp-bundles").unwrap_or_default(),
405        agents: cfg.get_map_keys("agents").unwrap_or_default(),
406    }
407}

420/// `GET /api/onboard/sections` — list every top-level config section.
421///
422/// Schema-driven: walks `Config::prop_fields()` and collects unique first
423/// segments, then asks `Config::map_key_sections()` for which ones have
424/// pickers. The 4 onboarding sections (`model_providers`, `channels`, `memory`,
425/// `tunnel`) keep their existing per-section dispatch in
426/// `handle_section_picker`; everything else (`gateway`, `observability`,
427/// `scheduler`, ...) renders as a direct form. Adding a new top-level
428/// field to `Config` makes it appear here automatically.
429pub async fn handle_sections(State(state): State<AppState>, headers: HeaderMap) -> Response {
430    if let Err(e) = require_auth(&state, &headers) {
431        return e.into_response();
432    }
433    let cfg = state.config.read().clone();
434    let completed: std::collections::HashSet<String> = cfg
435        .onboard_state
436        .completed_sections
437        .iter()
438        .cloned()
439        .collect();
440
441    // First segment of every reachable prop path. BTreeSet for stable
442    // alphabetical order and dedup.
443    let mut roots: std::collections::BTreeSet<String> = cfg
444        .prop_fields()
445        .iter()
446        .filter_map(|f| f.name.split('.').next().map(str::to_string))
447        .collect();
448
449    // System / housekeeping fields the user never edits via the dashboard.
450    for hidden in HIDDEN_TOP_LEVEL {
451        roots.remove(*hidden);
452    }
453
454    // A section gets a picker only when its OWN root carries a map (path
455    // == key) or its immediate child is a typed-family map (path == key
456    // + "." + one segment). Deeper nested maps belong to a subsection's
457    // own editor and must not promote their top-level section to a
458    // picker — `cost.rates.providers.models.<type>` is the rate-sheet's
459    // concern, not a reason to give `[cost]` an Add affordance.
460    let all_map_paths: Vec<&'static str> = zeroclaw_config::schema::Config::map_key_sections()
461        .iter()
462        .map(|s| s.path)
463        .collect();
464    let section_has_picker_for_key = |key: &str| -> bool {
465        let key_dot = format!("{key}.");
466        all_map_paths.iter().any(|p| {
467            *p == key
468                || p.strip_prefix(&key_dot)
469                    .is_some_and(|rest| !rest.contains('.'))
470        })
471    };
472
473    // Ensure map-keyed sections surface as sidebar entries even when their
474    // HashMap is empty (prop_fields() only yields paths for populated
475    // entries). First segments only — the prefix-dedup pass below drops
476    // bare parent segments when a multi-segment child is present.
477    let map_keyed_roots: std::collections::HashSet<&'static str> = all_map_paths
478        .iter()
479        .filter_map(|p| p.split('.').next())
480        .collect();
481    for &prefix in &map_keyed_roots {
482        roots.insert(prefix.to_string());
483    }
484
485    // Synthetic onboarding sections — keys that aren't fields on Config
486    // but are part of the wizard flow (personality lives as markdown
487    // files, not TOML). Inject so the canonical-order sort places them
488    // correctly and frontends don't need to know which ones to splice.
489    for s in zeroclaw_config::sections::ONBOARDING_SECTIONS {
490        roots.insert(s.as_str().to_string());
491    }
492
493    // Drop bare parent-segment entries when a dotted child is present
494    // — `providers` is phantom once `providers.models` etc. are listed.
495    let prefixes_with_children: std::collections::HashSet<String> = roots
496        .iter()
497        .filter_map(|k| k.split_once('.').map(|(parent, _)| parent.to_string()))
498        .collect();
499    roots.retain(|k| k.contains('.') || !prefixes_with_children.contains(k));
500
501    // Hard-ban the rate-sheet subtree from the sidebar. `[cost.rates.*]` is
502    // edited from inside the `[cost]` section's tabs (and from each
503    // provider-type page's Costs tab); it has no standalone picker, no
504    // direct form at any intermediate depth, and surfacing a path like
505    // `cost.rates.providers.tts` as its own sidebar entry only yields a
506    // dead-end "no picker" page.
507    roots.retain(|k| !k.starts_with("cost.rates"));
508
509    // Sort: onboarding-wizard sections first in their canonical order
510    // (single source of truth in `zeroclaw_config::sections`), then
511    // everything else alphabetically. This is what makes /onboard's wizard
512    // order and /config's foundation grouping derive from one Rust const
513    // — frontends consume the response order directly.
514    let mut ordered: Vec<String> = roots.into_iter().collect();
515    ordered.sort_by(|a, b| {
516        match (
517            zeroclaw_config::sections::section_index_for_key(a),
518            zeroclaw_config::sections::section_index_for_key(b),
519        ) {
520            (Some(ai), Some(bi)) => ai.cmp(&bi),
521            (Some(_), None) => std::cmp::Ordering::Less,
522            (None, Some(_)) => std::cmp::Ordering::Greater,
523            (None, None) => a.cmp(b),
524        }
525    });
526
527    let sections: Vec<SectionInfo> = ordered
528        .into_iter()
529        .map(|key| {
530            // Picker eligibility = anything `handle_section_picker`
531            // dispatches non-trivially. Wizard sections that opt out
532            // (workspace/hardware/personality) are direct-form. Map-keyed
533            // sections outside the wizard (multi-agent peer groups, etc.)
534            // get the generic schema-walk picker.
535            let wizard = zeroclaw_config::sections::Section::from_key(&key);
536            let has_picker = match wizard {
537                Some(w) => !matches!(
538                    w,
539                    zeroclaw_config::sections::Section::Hardware
540                        | zeroclaw_config::sections::Section::Mcp
541                        | zeroclaw_config::sections::Section::Skills
542                ),
543                None => section_has_picker_for_key(&key),
544            };
545            SectionInfo {
546                completed: completed.contains(&key),
547                ready: section_ready(&cfg, &key, completed.contains(&key)),
548                label: humanize_section(&key),
549                help: section_help(&key).to_string(),
550                has_picker,
551                group: section_group(&key).to_string(),
552                is_onboarding: wizard.is_some(),
553                shape: wizard.map(zeroclaw_config::sections::Section::shape),
554                key,
555            }
556        })
557        .collect();
558
559    axum::Json(SectionsResponse { sections }).into_response()
560}

562fn section_ready(cfg: &zeroclaw_config::schema::Config, key: &str, completed_marker: bool) -> bool {
563    use zeroclaw_config::sections::Section;
564    match Section::from_key(key) {
565        Some(Section::ModelProviders) => any_usable_model_provider(cfg),
566        Some(Section::RiskProfiles) => !cfg.risk_profiles.is_empty(),
567        Some(Section::RuntimeProfiles) => !cfg.runtime_profiles.is_empty(),
568        Some(Section::Storage) => cfg
569            .prop_fields()
570            .iter()
571            .any(|field| field.name.starts_with("storage.")),
572        Some(Section::Memory) => completed_marker,
573        Some(Section::Agents) => cfg
574            .agents

701/// `GET /api/onboard/sections/<section>` — picker items for that section.
702///
703/// Per-section dispatch:
704/// * `providers` → `zeroclaw_providers::list_model_providers()` (CLI's catalog).
705/// * `memory` → `zeroclaw_memory::selectable_memory_backends()`.
706/// * `channels` / `tunnel` → schema-walk: clone config, `init_defaults` the
707///   section, then strip the section prefix from `prop_fields()` and dedupe
708///   by first segment. Same trick the TUI uses; new channels appear
709///   automatically when a `#[nested] Option<...>` field is added.
710/// * Anything else returns 404 (hardware has no picker).
711pub async fn handle_section_picker(
712    State(state): State<AppState>,
713    headers: HeaderMap,
714    axum::extract::Path(SectionPath { section }): axum::extract::Path<SectionPath>,
715) -> Response {
716    if let Err(e) = require_auth(&state, &headers) {
717        return e.into_response();
718    }
719    let cfg = state.config.read().clone();
720
721    use zeroclaw_config::sections::Section;
722    let Some(section_enum) = Section::from_key(&section) else {
723        return error_response(
724            ConfigApiError::new(
725                ConfigApiCode::PathNotFound,
726                format!(
727                    "section `{section}` has no picker; render its fields \
728                     via GET /api/config/list?prefix={section}"
729                ),
730            )
731            .with_path(section.as_str()),
732        );
733    };
734    let help = section_help(section_enum.as_str()).to_string();
735    let items = match picker_items_for(section_enum, &cfg) {
736        PickerDispatch::Items(items) => items,
737        PickerDispatch::DirectForm => {
738            return error_response(
739                ConfigApiError::new(
740                    ConfigApiCode::PathNotFound,
741                    format!(
742                        "section `{section_enum}` is a direct-form section with no picker; \
743                         render fields via GET /api/config/list?prefix={section_enum}"
744                    ),
745                )
746                .with_path(section_enum.as_str()),
747            );
748        }
749    };
750
751    axum::Json(PickerResponse {
752        section,
753        items,
754        help,
755    })
756    .into_response()
757}

759/// Result of picker dispatch for a [`Section`]. `Items` carries the
760/// list rendered into the dashboard / CLI picker UI; `DirectForm`
761/// signals a section without a picker step (the caller falls through
762/// to `/api/config/list?prefix=<section>` for direct field rendering).
763///
764/// Splitting this out from `handle_section_picker` keeps the per-Section
765/// dispatch a pure function — testable without an `AppState` mock and
766/// exhaustively coverable by iterating every variant.
767enum PickerDispatch {
768    Items(Vec<PickerItem>),
769    DirectForm,
770}

772/// Per-section picker dispatch. Exhaustive over [`Section`] so adding a
773/// variant fails to compile until it gets a routing arm. The DRY
774/// version of what the dashboard's per-section view boils down to.
775fn picker_items_for(
776    section: zeroclaw_config::sections::Section,
777    cfg: &zeroclaw_config::schema::Config,
778) -> PickerDispatch {
779    use zeroclaw_config::sections::Section;
780    match section {
781        Section::ModelProviders => PickerDispatch::Items(providers_picker(cfg)),
782        // TTS / transcription share the typed-family two-tier shape. Each
783        // family enumerates its picker via `schema_walk_picker(<family>)`
784        // — the same machinery channels uses, so no per-section catalog
785        // table to drift.
786        Section::TtsProviders | Section::TranscriptionProviders => {
787            PickerDispatch::Items(schema_walk_picker(cfg, section.as_str()))
788        }
789        Section::Memory => PickerDispatch::Items(memory_picker(cfg)),
790        Section::Channels => PickerDispatch::Items(schema_walk_picker(cfg, "channels")),
791        Section::Tunnel => PickerDispatch::Items(schema_walk_picker_with_none(
792            cfg,
793            "tunnel",
794            "tunnel.tunnel-provider",
795        )),
796        Section::Agents => PickerDispatch::Items(agents_picker(cfg)),
797        // Storage is two-tier (`storage.<kind>.<alias>`) — same shape
798        // and walker as channels and the typed-provider families.
799        Section::Storage => PickerDispatch::Items(storage_picker(cfg)),
800        // OneTierAliasMap explorer sections: pick a key from the live
801        // HashMap. Generic walker covers every section whose schema is
802        // `<section>.<alias>` (operator-named keys, no closed kind set).
803        Section::PeerGroups
804        | Section::Cron
805        | Section::McpBundles
806        | Section::KnowledgeBundles
807        | Section::SkillBundles
808        | Section::RiskProfiles
809        | Section::RuntimeProfiles => {
810            PickerDispatch::Items(one_tier_alias_map_picker(cfg, section.as_str()))
811        }
812        Section::Hardware | Section::Mcp | Section::Skills => PickerDispatch::DirectForm,
813    }
814}

959/// Generic schema-walk picker for sections like `channels` whose subsections
960/// are `#[nested] HashMap<String, T>` fields. Discovery: use `map_key_sections()`
961/// to enumerate all statically-known sub-sections under `<section>.` — this
962/// works for HashMap-based channels without needing init_defaults to insert
963/// entries (HashMap fields start empty and init_defaults leaves them empty).
964fn schema_walk_picker(cfg: &zeroclaw_config::schema::Config, section: &str) -> Vec<PickerItem> {
965    let prefix_with_dot = format!("{section}.");
966
967    // Configured: any alias present on this type (has at least one entry in its HashMap).
968    let configured: std::collections::BTreeSet<String> = cfg
969        .prop_fields()
970        .iter()
971        .filter_map(|f| f.name.strip_prefix(&prefix_with_dot))
972        .filter_map(|suffix| suffix.split_once('.').map(|(head, _)| head.to_string()))
973        .collect();
974
975    // All known channel/section types from schema metadata — statically known,
976    // no HashMap entries needed.
977    let all: std::collections::BTreeSet<String> =
978        zeroclaw_config::schema::Config::map_key_sections()
979            .into_iter()
980            .filter_map(|s| {
981                s.path
982                    .strip_prefix(&prefix_with_dot)
983                    .filter(|rest| !rest.contains('.'))
984                    .map(String::from)
985            })
986            .collect();
987
988    all.into_iter()
989        .map(|name| {
990            // Channel configs no longer carry an `enabled` field; a channel is
991            // active when an enabled agent references it. Badge = "configured" when
992            // at least one alias exists, absent otherwise.
993            let badge = if configured.contains(&name) {
994                Some("configured".to_string())
995            } else {
996                None
997            };
998            PickerItem {
999                key: name.clone(),
1000                label: name.clone(),
1001                description: None,
1002                badge,
1003            }
1004        })
1005        .collect()
1006}

1008/// Generic picker for `OneTierAliasMap` sections — walks the live
1009/// `prop_fields()` for the section prefix and returns one PickerItem
1010/// per operator-defined alias. The closed-kind enumeration that
1011/// [`schema_walk_picker`] does via `Config::map_key_sections()` doesn't
1012/// apply here: aliases under `peer_groups`, `cron`, `risk_profiles`,
1013/// etc. are operator-named, with no statically-known catalog. Every
1014/// existing alias is reported `configured`; the dashboard's `+ Add`
1015/// affordance handles new-key creation through
1016/// [`handle_select_item`].
1017fn one_tier_alias_map_picker(
1018    cfg: &zeroclaw_config::schema::Config,
1019    section: &str,
1020) -> Vec<PickerItem> {
1021    let prefix_with_dot = format!("{section}.");
1022    let mut keys: std::collections::BTreeSet<String> = std::collections::BTreeSet::new();
1023    for field in cfg.prop_fields() {
1024        let Some(suffix) = field.name.strip_prefix(&prefix_with_dot) else {
1025            continue;
1026        };
1027        let head = suffix.split_once('.').map_or(suffix, |(h, _)| h);
1028        if head.is_empty() {
1029            continue;
1030        }
1031        keys.insert(head.to_string());
1032    }
1033    keys.into_iter()
1034        .map(|key| PickerItem {
1035            key: key.clone(),
1036            label: key,
1037            description: None,
1038            badge: Some("configured".to_string()),
1039        })
1040        .collect()
1041}

1183    if matches!(
1184        section,
1185        zeroclaw_config::sections::Section::ModelProviders
1186            | zeroclaw_config::sections::Section::TtsProviders
1187            | zeroclaw_config::sections::Section::TranscriptionProviders

1190        // (`atomic_chat`, `local_whisper`), while Configurable field paths

1192        // snake_case; normalize only the config-map path used to create and
1193        // edit the alias block.
1194        key.replace('_', "-")
1195    } else {
1196        key.to_string()
1197    }

1200pub async fn handle_section_select(
1201    State(state): State<AppState>,
1202    headers: HeaderMap,
1203    axum::extract::Path(SectionItemPath { section, key }): axum::extract::Path<SectionItemPath>,
1204    body: Option<axum::extract::Json<SectionSelectBody>>,
1205) -> Response {
1206    if let Err(e) = require_auth(&state, &headers) {
1207        return e.into_response();
1208    }
1209
1210    let alias = body
1211        .and_then(|b| b.0.alias)
1212        .map(|s| s.trim().to_string())
1213        .filter(|s| !s.is_empty())
1214        .unwrap_or_else(|| "default".to_string());
1215
1216    let mut working = state.config.read().clone();
1217
1218    use zeroclaw_config::sections::Section;
1219    let Some(section_enum) = Section::from_key(&section) else {
1220        return error_response(
1221            ConfigApiError::new(
1222                ConfigApiCode::PathNotFound,
1223                format!("no picker semantics defined for section `{section}`"),
1224            )
1225            .with_path(section.as_str()),
1226        );
1227    };
1228
1229    let (fields_prefix, created) = match section_enum {
1230        Section::ModelProviders | Section::TtsProviders | Section::TranscriptionProviders => {
1231            // Two-tier typed-family path: outer bucket is the family
1232            // (`model_providers.<type>` etc.), inner key is the alias the
1233            // operator named. `create_map_key` is idempotent so re-selecting
1234            // an existing type/alias is a no-op for the bucket and just
1235            // returns the form prefix for the alias.
1236            let family = section_enum.as_str();
1237            let config_key = typed_family_config_key(section_enum, &key);
1238            let created = working
1239                .create_map_key(&format!("{family}.{config_key}"), &alias)
1240                .map_err(|msg| {
1241                    error_response(
1242                        ConfigApiError::new(
1243                            ConfigApiCode::PathNotFound,
1244                            format!("could not select {family} `{key}` alias `{alias}`: {msg}"),
1245                        )
1246                        .with_path(format!("{family}.{config_key}")),
1247                    )
1248                });
1249            let created = match created {
1250                Ok(c) => c,
1251                Err(resp) => return resp,
1252            };
1253            // Per-family typed configs derive their own default endpoint
1254            // URI via family traits at runtime construction time.
1255            (format!("{family}.{config_key}.{alias}"), created)
1256        }
1257        Section::Channels => {
1258            let created = working
1259                .create_map_key(&format!("channels.{key}"), &alias)
1260                .map_err(|msg| {
1261                    error_response(
1262                        ConfigApiError::new(
1263                            ConfigApiCode::PathNotFound,
1264                            format!("could not select channel `{key}` alias `{alias}`: {msg}"),
1265                        )
1266                        .with_path(format!("channels.{key}")),
1267                    )
1268                });
1269            let created = match created {
1270                Ok(c) => c,
1271                Err(resp) => return resp,
1272            };
1273            // The per-channel-type struct's `enabled` field defaults to
1274            // `false` (pre-v0.8.0 paste-safety rationale: don't fire a
1275            // listener on a half-pasted block). For wizard-driven creation
1276            // the operator has just consciously added the alias, so flip
1277            // the new entry's `enabled` to true. Re-selecting an existing
1278            // alias is a no-op (created=false), so user-edited values are
1279            // never trampled.
1280            if created {
1281                let enabled_path = format!("channels.{key}.{alias}.enabled");
1282                if let Err(e) = working.set_prop_persistent(&enabled_path, "true") {
1283                    ::zeroclaw_log::record!(
1284                        WARN,
1285                        ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note)
1286                            .with_outcome(::zeroclaw_log::EventOutcome::Unknown)
1287                            .with_attrs(
1288                                ::serde_json::json!({"path": enabled_path, "error": format!("{}", e)})
1289                            ),
1290                        "failed to default-enable newly created channel; operator must toggle manually"
1291                    );
1292                }
1293            }
1294            (format!("channels.{key}.{alias}"), created)
1295        }
1296        Section::Agents
1297        | Section::PeerGroups
1298        | Section::Cron
1299        | Section::McpBundles
1300        | Section::KnowledgeBundles
1301        | Section::SkillBundles
1302        | Section::RiskProfiles
1303        | Section::RuntimeProfiles => {
1304            // OneTierAliasMap: the URL path key IS the alias. One
1305            // `create_map_key("<section>", &key)` call works for every
1306            // operator-named HashMap section; create_map_key is
1307            // idempotent, so selecting an existing alias just returns
1308            // the form prefix without modifying anything.
1309            let section_key = section_enum.as_str();
1310            let created = working.create_map_key(section_key, &key).map_err(|msg| {
1311                error_response(
1312                    ConfigApiError::new(
1313                        ConfigApiCode::PathNotFound,
1314                        format!("could not select {section_key} alias `{key}`: {msg}"),
1315                    )
1316                    .with_path(section_key),
1317                )
1318            });
1319            let created = match created {
1320                Ok(c) => c,
1321                Err(resp) => return resp,
1322            };
1323            // Agents need a per-alias workspace dir on disk so the
1324            // PersonalityEditor and the runtime have somewhere to read
1325            // and write IDENTITY.md / SOUL.md / USER.md / etc.
1326            if created && matches!(section_enum, Section::Agents) {
1327                apply_first_run_agent_defaults(&mut working, &key);
1328                let workspace_dir = working.agent_workspace_dir(&key);
1329                if let Err(err) = tokio::fs::create_dir_all(&workspace_dir).await {
1330                    return error_response(
1331                        ConfigApiError::new(
1332                            ConfigApiCode::ValidationFailed,
1333                            format!(
1334                                "created agent `{key}` but failed to scaffold workspace at {}: {err}",
1335                                workspace_dir.display()
1336                            ),
1337                        )
1338                        .with_path(section_key),
1339                    );
1340                }
1341                if let Err(err) =
1342                    zeroclaw_config::schema::ensure_bootstrap_files(&workspace_dir).await
1343                {
1344                    ::zeroclaw_log::record!(WARN, ::zeroclaw_log::Event::new(module_path!(), ::zeroclaw_log::Action::Note).with_outcome(::zeroclaw_log::EventOutcome::Unknown).with_attrs(::serde_json::json!({"agent": key, "workspace": workspace_dir.display().to_string(), "err": err.to_string()})), "agent workspace scaffolded but bootstrap files seed failed (continuing)");
1345                }
1346            }
1347            (format!("{section_key}.{key}"), created)
1348        }
1349        Section::Storage => {
1350            // Two-tier typed-family (`storage.<kind>.<alias>`) — same
1351            // shape and selection flow as model_providers / tts_providers /
1352            // transcription_providers. Outer bucket is the storage kind
1353            // (sqlite, postgres, qdrant, markdown, lucid); inner key is
1354            // the operator-named alias.
1355            let created = working
1356                .create_map_key(&format!("storage.{key}"), &alias)
1357                .map_err(|msg| {
1358                    error_response(
1359                        ConfigApiError::new(
1360                            ConfigApiCode::PathNotFound,
1361                            format!("could not select storage `{key}` alias `{alias}`: {msg}"),
1362                        )
1363                        .with_path(format!("storage.{key}")),
1364                    )
1365                });
1366            let created = match created {
1367                Ok(c) => c,
1368                Err(resp) => return resp,
1369            };
1370            mark_onboard_section_completed(&mut working, "storage");
1371            (format!("storage.{key}.{alias}"), created)
1372        }
1373        Section::Memory => {
1374            // Set memory.backend to the picked key. Fields_prefix points at
1375            // `memory` so the form renders the whole memory section
1376            // (the active backend's specific fields show up there).
1377            if let Err(e) = working.set_prop_persistent("memory.backend", &key) {
1378                return error_response(
1379                    ConfigApiError::new(
1380                        ConfigApiCode::ValidationFailed,
1381                        format!("could not set memory.backend = `{key}`: {e}"),
1382                    )
1383                    .with_path("memory.backend"),
1384                );
1385            }
1386            mark_onboard_section_completed(&mut working, "memory");
1387            ("memory".to_string(), true)
1388        }
1389        Section::Tunnel => {
1390            if let Err(e) = working.set_prop_persistent("tunnel.tunnel-provider", &key) {
1391                return error_response(
1392                    ConfigApiError::new(
1393                        ConfigApiCode::ValidationFailed,
1394                        format!("could not set tunnel.tunnel-provider = `{key}`: {e}"),
1395                    )
1396                    .with_path("tunnel.tunnel-provider"),
1397                );
1398            }
1399            let prefix = if key == "none" {
1400                "tunnel".to_string()
1401            } else {
1402                let p = format!("tunnel.{key}");
1403                working.init_defaults(Some(&p));
1404                p
1405            };
1406            (prefix, true)
1407        }
1408        Section::Hardware | Section::Mcp | Section::Skills => {
1409            return error_response(
1410                ConfigApiError::new(
1411                    ConfigApiCode::PathNotFound,
1412                    format!(
1413                        "section `{}` is a direct-form section with no picker; \
1414                         render fields via GET /api/config/list?prefix={}",
1415                        section_enum, section_enum
1416                    ),
1417                )
1418                .with_path(section_enum.as_str()),
1419            );
1420        }
1421    };
1422
1423    if created {
1424        working.mark_dirty(&fields_prefix);
1425    }
1426
1427    if let Err(e) = working.save_dirty().await {
1428        return error_response(ConfigApiError::new(
1429            ConfigApiCode::ReloadFailed,
1430            format!("save after select failed: {e}"),
1431        ));
1432    }
1433    *state.config.write() = working;
1434
1435    axum::Json(SelectItemResponse {
1436        fields_prefix,
1437        created,
1438    })
1439    .into_response()
1440}
