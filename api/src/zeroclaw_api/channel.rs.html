33/// A message received from or sent to a channel
34#[derive(Debug, Clone)]
35pub struct ChannelMessage {
36    pub id: String,
37    pub sender: String,
38    pub reply_target: String,
39    pub content: String,
40    pub channel: String,
41    /// ZeroClaw channel alias (the `<alias>` half of `[channels.<type>.<alias>]`)
42    /// when the platform supports multiple bot instances. Used by
43    /// session_key construction so two bots on the same platform compute
44    /// distinct session IDs and don't share conversation history. `None`
45    /// for channels that don't have an alias concept yet (webhook, cli).
46    pub channel_alias: Option<String>,
47    pub timestamp: u64,
48    /// Platform thread identifier (e.g. Slack `ts`, Discord thread ID).
49    /// When set, replies should be posted as threaded responses.
50    pub thread_ts: Option<String>,
51    /// Thread scope identifier for interruption/cancellation grouping.
52    /// Distinct from `thread_ts` (reply anchor): this is `Some` only when the message
53    /// is genuinely inside a reply thread and should be isolated from other threads.
54    /// `None` means top-level — scope is sender+channel only.
55    pub interruption_scope_id: Option<String>,
56    /// Media attachments (audio, images, video) for the media pipeline.
57    /// Channels populate this when they receive media alongside a text message.
58    /// Defaults to empty — existing channels are unaffected.
59    pub attachments: Vec<MediaAttachment>,
60}

62/// Message to send through a channel
63#[derive(Debug, Clone)]
64pub struct SendMessage {
65    pub content: String,
66    pub recipient: String,
67    pub subject: Option<String>,
68    /// Platform thread identifier for threaded replies (e.g. Slack `thread_ts`).
69    pub thread_ts: Option<String>,
70    /// Optional cancellation token for interruptible delivery (e.g. multi-message mode).
71    pub cancellation_token: Option<CancellationToken>,
72    /// File attachments to send with the message.
73    /// Channels that don't support attachments ignore this field.
74    pub attachments: Vec<MediaAttachment>,
75}

125/// Core channel trait — implement for any messaging platform.
126///
127/// Every `Channel` is `Attributable`: the orchestrator's spawn site opens
128/// `attribution_span!(&*ch)` so log emissions from within `listen()` / `send()`
129/// inherit `channel = <type>.<alias>` from the trait object's role + alias.
130#[async_trait]
131pub trait Channel: Send + Sync + crate::attribution::Attributable {

140

143        true
144    }
145
146    /// Signal that the bot is processing a response (e.g. "typing" indicator).
147    async fn start_typing(&self, _recipient: &str) -> anyhow::Result<()> {
148        Ok(())
149    }
150
151    /// Stop any active typing indicator.
152    async fn stop_typing(&self, _recipient: &str) -> anyhow::Result<()> {
153        Ok(())
154    }
155
156    /// Whether this channel supports progressive message updates via draft edits.
157    fn supports_draft_updates(&self) -> bool {
158        false
159    }
160
161    /// Self-loop guard for multi-agent runs.
162    ///
163    /// Returns the bot's own handle/identity on this channel
164    /// (e.g. `@my_bot` for Telegram, the bot's user ID for Discord)
165    /// when known, so the orchestrator can drop inbound events whose
166    /// `sender` matches: a bot must never respond to its own
167    /// messages, even if a misconfigured peer group lists the bot's
168    /// handle as an external peer.
169    ///
170    /// **Channels that handle inbound traffic must override this.**
171    /// The default `None` makes both layers of the orchestrator's
172    /// self-loop guard (the SDK-side `drop_self_messages` here, and
173    /// the agent-loop fallback `peers::should_drop_self_loop`) into
174    /// no-ops — both layers consult the same `self_handle`, so a
175    /// channel that returns `None` has no protection from looping on
176    /// its own outbound. Outbound-only channels (webhook, gmail-push,
177    /// voice-call) never see inbound and can keep the default. The
178    /// in-tree overrides currently cover Telegram (`bot_username`
179    /// cache), IRC (configured nickname), Discord (decoded from token),
180    /// Slack (cached `auth.test` user_id); other inbound channels
181    /// remain on the default and rely on per-impl filtering instead
182    /// of the shared guard.
183    fn self_handle(&self) -> Option<String> {
184        None
185    }
186
187    /// The exact form the bot expects to see when addressed by users on
188    /// this channel. Discord wraps the snowflake as `<@1088...>`,
189    /// Telegram presents `@bot_username`, Matrix presents
190    /// `@bot:server`, Slack wraps the user ID as `<@U02...>`. Returned
191    /// verbatim into the per-channel system prompt so the agent
192    /// recognizes its own mention without guessing, and uses the same
193    /// form to tag itself or peers in outbound replies.
194    ///
195    /// Default `None` for channels that have no inbound mention
196    /// concept (CLI, webhook, hardware, ACP elicitation). Channels
197    /// that override `self_handle` should usually override this too,
198    /// applying their platform-native mention wrapper to the handle.
199    fn self_addressed_mention(&self) -> Option<String> {
200        None
201    }
202
203    /// Whether the orchestrator should drop an inbound message as
204    /// self-authored (multi-agent self-loop guard).
205    ///
206    /// Default implementation compares `msg.sender` against
207    /// [`Self::self_handle`] case-insensitively, after stripping a
208    /// leading `@` from each side so Telegram-style handles match
209    /// regardless of which form the SDK delivers. Override only for
210    /// platforms whose identity comparison is non-string (e.g. a
211    /// numeric Discord user ID is `as_str` already; this default
212    /// works there too).
213    fn drop_self_messages(&self, msg: &ChannelMessage) -> bool {
214        let Some(handle) = self.self_handle() else {
215            return false;
216        };
217        let handle_norm = handle.trim_start_matches('@').to_ascii_lowercase();
218        let sender_norm = msg.sender.trim_start_matches('@').to_ascii_lowercase();
219        !handle_norm.is_empty() && handle_norm == sender_norm
220    }
221
222    /// Whether this channel supports multi-message streaming delivery.
223    fn supports_multi_message_streaming(&self) -> bool {
224        false
225    }
226
227    /// Minimum delay (ms) between sending each paragraph in multi-message mode.
228    fn multi_message_delay_ms(&self) -> u64 {
229        800
230    }

294        Ok(())

299        Ok(())

303    async fn redact_message(
304        &self,
305        _channel_id: &str,

312    /// Request interactive tool-call approval from the channel operator.
313    ///
314    /// Returns `Ok(Some(response))` when the operator answers within the
315    /// channel's configured `approval_timeout_secs`; timeouts are surfaced
316    /// as `Deny`. Returns `Ok(None)` only for channels that do not implement
317    /// the prompt at all — the caller should fall back to its default policy
318    /// (typically auto-deny).
319    ///
320    /// Surface varies by channel:
321    /// - **Telegram** uses inline keyboard buttons.
322    /// - **Slack** Socket Mode uses Block Kit buttons; webhook fallback and
323    ///   non–Socket Mode deployments use a token text reply.
324    /// - **Discord, Signal, Matrix, WhatsApp** embed a 6-character
325    ///   alphanumeric token in the prompt and wait for a
326    ///   `<token> approve|deny|always` reply on the same conversation.
327    async fn request_approval(
328        &self,
329        _recipient: &str,
330        _request: &ChannelApprovalRequest,
331    ) -> anyhow::Result<Option<ChannelApprovalResponse>> {
332        Ok(None)
333    }

335    /// Ask the user a multiple-choice question and return the chosen option's text.
336    ///
337    /// Returns `Ok(Some(answer))` if the channel handled the question natively
338    /// (e.g. ACP `session/request_permission`, Telegram inline keyboard).
339    /// Returns `Ok(None)` to signal the caller should fall back to the
340    /// generic `send` + `listen` flow. Default impl returns `None`.
341    ///
342    /// Free-form questions (no choices) are not modeled here yet — they
343    /// require the ACP elicitation RFD to land for a clean cross-channel API.
344    async fn request_choice(
345        &self,
346        _question: &str,
347        _choices: &[String],
348        _timeout: std::time::Duration,
349    ) -> anyhow::Result<Option<String>> {
350        Ok(None)
351    }
352
353    /// Whether this channel can answer free-form (no-choices) `ask_user`
354    /// questions via the standard `send` + `listen` flow.
355    ///
356    /// Channels that can only handle structured choices (e.g. ACP today, until
357    /// the elicitation RFD lands) should return `false` so callers can fail
358    /// fast with a useful error instead of timing out on `listen`.
359    fn supports_free_form_ask(&self) -> bool {
360        true
361    }
